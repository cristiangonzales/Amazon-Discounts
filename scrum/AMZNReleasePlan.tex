	\section*{High Level Goals}
		\vspace{-3mm}
		\begin{itemize}
			\item Daemon that finds items discounted heavily (95-99\% or user specified).
			\item Daemon notifies user of items found.
			\item Daemon automatically purchases items.
		\end{itemize}
